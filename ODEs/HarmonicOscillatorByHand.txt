% dv/dt = ?x + sin(2t)
% dx/dt = v
% Dt = 0.1
% x(0) = 0;
% V(0) = 1;

v(i+1) = v(i) + Dt*(-x(i) + sin(2*t));
x(i+1) = x(i) + Dt*v(i);

x(1) = 0;
v(1) = 1;

v(2) = v(1) + 0.1*(-x(1) + sin(2*0.1)); % 1.0199
x(2) = x(1) + 0.1*v(1);                 % 0.1000

v(3) = v(2) + 0.1*(-x(2) + sin(2*0.2)); % 1.0488
x(3) = x(2) + 0.2*v(2);                 % 0.3040
